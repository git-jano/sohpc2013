    function ns/op() {
      local _s="${1}"
      local _m="${2}"
      if empty "${_m}" ; then
        _m="${ns_op}"
      fi

      local o
      for o in ${_m} ; do
        if test "${_s:0:1}" = "${o}" ; then
          $success
        fi
      done
      $failure
    }
